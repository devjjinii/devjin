## 시/분/초 (00:00:00)

* VARCHAR2
* 총 걸린시간 계산 (시:분:초)

```sql
SELECT
       CASE WHEN ERROR_END_DT IS NULL THEN '00:00:00'   -- 종료일이 NULL 일 때,
       ELSE
            CASE WHEN TRUNC((TO_DATE(ERROR_END_DT || UPDATE_TT, 'YYYYMMDDHH24MISS') - TO_DATE(ERROR_START_DT || CONTACTSTART_TT, 'YYYYMMDDHH24MISS'))*24) = '0' THEN '00:' ||
                  LPAD(TRUNC(MOD((TO_DATE(ERROR_END_DT || UPDATE_TT, 'YYYYMMDDHH24MISS') - TO_DATE(ERROR_START_DT || CONTACTSTART_TT, 'YYYYMMDDHH24MISS'))*24,1)*60),'2','0') || ':' ||
                  LPAD(TRUNC(ROUND(MOD((TO_DATE(ERROR_END_DT || UPDATE_TT, 'YYYYMMDDHH24MISS') - TO_DATE(ERROR_START_DT || CONTACTSTART_TT, 'YYYYMMDDHH24MISS'))*24*60,1)*60)),'2','0')
                ELSE
                    TRUNC((TO_DATE(ERROR_END_DT || UPDATE_TT, 'YYYYMMDDHH24MISS') - TO_DATE(ERROR_START_DT || CONTACTSTART_TT, 'YYYYMMDDHH24MISS'))*24) || ':' ||
                  LPAD(TRUNC(MOD((TO_DATE(ERROR_END_DT || UPDATE_TT, 'YYYYMMDDHH24MISS') - TO_DATE(ERROR_START_DT || CONTACTSTART_TT, 'YYYYMMDDHH24MISS'))*24,1)*60),'2','0') || ':' ||
                  LPAD(TRUNC(ROUND(MOD((TO_DATE(ERROR_END_DT || UPDATE_TT, 'YYYYMMDDHH24MISS') - TO_DATE(ERROR_START_DT || CONTACTSTART_TT, 'YYYYMMDDHH24MISS'))*24*60,1)*60)),'2','0')
            END

        END AS "TOTAL_TIME"

FROM TABLE;

```

* 예제 ( __일 00:00:00 시:분:초)
```sql
SELECT
    TRUNC(TO_DATE(ERROR_END_DT || UPDATE_TT, 'YYYYMMDDHH24MISS') - TO_DATE(ERROR_START_DT || CONTACTSTART_TT, 'YYYYMMDDHH24MISS')) || '일 ' ||
    LPAD(TRUNC(MOD((TO_DATE(ERROR_END_DT || UPDATE_TT, 'YYYYMMDDHH24MISS') - TO_DATE(ERROR_START_DT || CONTACTSTART_TT, 'YYYYMMDDHH24MISS')),1)*24),'2','0') || ':' ||
    LPAD(TRUNC(MOD((TO_DATE(ERROR_END_DT || UPDATE_TT, 'YYYYMMDDHH24MISS') - TO_DATE(ERROR_START_DT || CONTACTSTART_TT, 'YYYYMMDDHH24MISS'))*24,1)*60),'2','0') || ':' ||
    LPAD(TRUNC(ROUND(MOD((TO_DATE(ERROR_END_DT || UPDATE_TT, 'YYYYMMDDHH24MISS') - TO_DATE(ERROR_START_DT || CONTACTSTART_TT, 'YYYYMMDDHH24MISS'))*24*60,1)*60)),'2','0') AS "TOTAL_TIME"
FROM TABLE;
```
